async function traducirTexto() { const texto = document.getElementById("inputText").value; const source = document.getElementById("sourceLang").value; const target = document.getElementById("targetLang").value; const output = document.getElementById("output"); output.innerText = "Traduciendo..."; try { const res = await fetch("https://libretranslate.de/translate", { method: "POST", headers: { "Content-Type": "application/json" }, body: JSON.stringify({ q: texto, source: source, target: target, format: "text" }) }); const data = await res.json(); output.innerText = data.translatedText; } catch (error) { output.innerText = "Error en la traducción. Intenta de nuevo."; } }
